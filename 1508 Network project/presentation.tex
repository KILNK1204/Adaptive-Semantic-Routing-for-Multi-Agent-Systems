% Slide 2: Problem Statement
\begin{frame}
\frametitle{The Challenge}
\begin{itemize}
    \item Multi-agent LLM systems leverage specialized domain expertise
    \pause
    \item But they introduce a critical problem: \textbf{query routing}
    \pause
    \item Which agent should handle this query?
    \pause
    \item Trade-offs:
    \begin{itemize}
        \item \textbf{Accuracy}: Route to the right agent
        \item \textbf{Efficiency}: Minimize routing overhead
        \item \textbf{Adaptability}: Improve over time
    \end{itemize}
\end{itemize}
\end{frame}

% Slide 4: Method 1 - Semantic Similarity
\begin{frame}
\frametitle{Method 1: Semantic Similarity Baseline}
\begin{center}
    \includegraphics[width=0.6\textwidth]{method1_diagram.png}
\end{center}
\begin{itemize}
    \item Use pre-trained sentence embeddings (all-MiniLM-L6-v2)
    \item Encode query and agent descriptions
    \item Route to agent with highest cosine similarity
    \pause
    \item \textbf{Advantages}: Training-free, fast, stable
    \item \textbf{Disadvantage}: No learning from history
\end{itemize}
\end{frame}

% Slide 5: Method 2 - Performance-Weighted Routing
\begin{frame}
\frametitle{Method 2: Performance-Weighted Adaptive Routing}
\begin{equation*}
S_a(q) = 0.20 \cdot \text{semantic}(q,a) + 0.35 \cdot \text{accuracy}_a + 0.35 \cdot \text{indomain}_a + 0.10 \cdot \text{latency\_score}_a
\end{equation*}

\begin{itemize}
    \item Maintains per-agent statistics:
    \begin{itemize}
        \item Accuracy rate (from ground truth)
        \item IN\_DOMAIN rate (agent self-assessment)
        \item Average latency
    \end{itemize}
    \pause
    \item Updates via exponential moving average ($\alpha = 0.1$)
    \pause
    \item Learns during both training \textit{and} testing phases
\end{itemize}
\end{frame}

% Slide 6: Method 3 - LLM Meta-Classification
\begin{frame}
\frametitle{Method 3: LLM Meta-Classification}
\begin{itemize}
    \item Use another LLM call to route queries
    \pause
    \item Meta-classifier sees:
    \begin{itemize}
        \item All four agent descriptions
        \item The user's query
    \end{itemize}
    \pause
    \item LLM decides which agent is best
    \pause
    \item \textbf{Advantage}: Can reason about complex boundaries
    \item \textbf{Disadvantage}: Extra LLM call = high latency
\end{itemize}
\end{frame}

% Slide 7: Experimental Setup
\begin{frame}
\frametitle{Evaluation Design}
\begin{itemize}
    \item \textbf{Datasets}:
    \begin{itemize}
        \item Training: 120 labeled queries (30 per agent)
        \item Testing: 200 labeled queries
    \end{itemize}
    \pause
    \item \textbf{Reproducibility}: 5 complete trials with different random seeds (22, 32, 52, 62, 82)
    \pause
    \item \textbf{Quality Validation}: Spot-check 20 sampled responses
    \pause
    \item \textbf{Metrics}:
    \begin{itemize}
        \item Routing accuracy, in-domain rate
        \item Latency (routing + total)
        \item Answer quality (5-point scale)
    \end{itemize}
\end{itemize}
\end{frame}

% Slide 12: Key Insight
\begin{frame}
\frametitle{Key Innovation: Dual-Signal Routing}
\begin{itemize}
    \item \textbf{External signal}: Accuracy (ground truth)
    \begin{itemize}
        \item What the data says
    \end{itemize}
    \pause
    \item \textbf{Internal signal}: IN\_DOMAIN (agent self-assessment)
    \begin{itemize}
        \item What the agent believes about its capabilities
    \end{itemize}
    \pause
    \item \textbf{Finding}: Combining both signals (50-50 weight) works better than either alone
    \pause
    \item Method 2's 94.2\% in-domain rate shows agents are well-calibrated
    \item Enables learning without ground-truth labels during deployment!
\end{itemize}
\end{frame}

% Slide 13: Discussion
\begin{frame}
\frametitle{Why Method 2 Works}
\begin{itemize}
    \item Continuous learning: improves as it routes queries
    \pause
    \item Balanced approach: combines signals
    \begin{itemize}
        \item 20\% semantic relevance
        \item 35\% accuracy from history
        \item 35\% agent confidence
        \item 10\% latency
    \end{itemize}
    \pause
    \item Practical deployment: doesn't require perfect routing
    \begin{itemize}
        \item 72.8\% correct routing yields 4.50/5.0 quality
        \item 27.2\% misroutes still acceptable
    \end{itemize}
\end{itemize}
\end{frame}
